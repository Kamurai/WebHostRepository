Create Table Race(
	Race		Char(12)	Not Null,
	CONSTRAINT	RacePK		Primary Key(Race),
	BigGuyLimit	Number(1)	Default 1
	CONSTRAINT	CkBigGuyLimit	Check(BigGuyLimit > 0 and BigGuyLimit < 9)
	);

Create Table Position(
	PositionId	Number(2)	Not Null,
	Race		Char(12)	Not Null,
	PositionName	Char(15)	Not Null,
	Cost		Number(6)	Not Null,
	Movement	Number(2)	Not Null,
	Strength	Number(2)	Not Null,
	Agility		Number(2)	Not Null,
	Armorvalue	Number(2)	Not Null,
	Limit		Number(2)	Default 16,
	BigGuy		Number(1)	Default 0,
	CONSTRAINT	PositionIdPK	Primary Key(PositionId),
	CONSTRAINT	CkPositionId	Check(PositionId > 0 and PositionId < 100),
	CONSTRAINT	PositionRaceFK	Foreign Key(Race)
	REFERENCES	Race(Race),
	CONSTRAINT	CkCost		Check(Cost > 9999),
	CONSTRAINT	CkMovement	Check(Movement > 0 and Movement <= 10),
	CONSTRAINT	CkStrength	Check(Strength > 0 and Strength <= 10),
	CONSTRAINT	CkAgility	Check(Agility > 0 and Agility <= 10),
	CONSTRAINT	CkArmorValue	Check(ArmorValue > 0 and ArmorValue <= 10),
	CONSTRAINT	CkLimit		Check(Limit >= 1 and Limit <= 16),
	CONSTRAINT	CkBigGuy	Check(BigGuy >= 0 and BigGuy <= 1)
	);

Create Table Manager(
	ManagerId	Number(5)	Not Null,
	ManagerName	Char(30)	Not Null,
	ManagerEmail	Char(40)	Not Null,
	ManagerAccount	Number(9)	default 1000000 Not Null,
	CONSTRAINT	ManagerIdPK	Primary Key(ManagerId),
	CONSTRAINT	ManagerNameU	Unique(ManagerName),
	CONSTRAINT	ManagerEmailU	Unique(ManagerEmail),
	CONSTRAINT	CkAccount	Check(ManagerAccount >= 0)
	);

Create Table Team(
	TeamId		Number(2)	Not Null,
	TeamName	Char(30)	Not Null,
	Race		Char(12)	Not Null,
	TeamColor1	Char(15)	Not Null,
	TeamColor2	Char(15)	Not Null,
	CONSTRAINT	TeamIdPK	Primary Key(TeamId),
	CONSTRAINT	TeamRaceFK	Foreign Key(Race)
	REFERENCES	Race(Race),
	CONSTRAINT	CkTeamId	Check(TeamId > 0 and TeamId < 100),
	CONSTRAINT	TeamNameU	Unique(TeamName)
	);

Create Table Player(
	PlayerId	Number(5)	Not Null,
	PlayerName	Char(15)	Not Null,
	PositionId	Number(2)	Not Null,
	TeamId		Number(5)	Not Null,
	ManagerId	Number(5)	Not Null,
	CONSTRAINT	PlayerIdPK	Primary Key(PlayerId),
	CONSTRAINT	PlayerNameU	Unique(PlayerName),
	CONSTRAINT	PositionIdFK	Foreign Key(PositionId)
	REFERENCES	Position(PositionId),
	CONSTRAINT	ManagerIdFK	Foreign Key(ManagerId)
	REFERENCES	Manager(ManagerId),
	CONSTRAINT	TeamIdFK	Foreign Key(TeamId)
	REFERENCES	Team(TeamId)
	);

create view PlayerView As
	select PlayerName, Position.Race, PositionName, TeamName, ManagerName from player, position, team, manager where player.teamid = team.teamid and player.managerid = manager.managerid and player.positionid = position.positionid order by player.playerid;

create view TeamView As
	select TeamName, Race, TeamColor1, TeamColor2 from Team order by TeamId;

create view PositionView As
	select Race, PositionName, Cost, Movement, Strength, Agility, ArmorValue from Position order by PositionId;

create view Managerview As
	select ManagerName, ManagerEmail, ManagerAccount from Manager order by ManagerId;





Create or replace procedure deleteplayer(
	targetplayer in number
	) IS
	BEGIN
		delete from player where playerid = targetplayer;
	END;
	/

Create or replace procedure changemanager(
	targetplayer in number,
	newmanager in number
	) IS
	BEGIN
		update player set managerid = newmanager where playerid = targetplayer;
	END;
	/


Create or replace procedure swapmanagers(
	player1 in number,
	player2 in number
	) AS
	manager1 int(5);
	manager2 int(5);
	cursor cursor1 is
		select managerid from player where playerid = player1;
	cursor cursor2 is
		select managerid from player where playerid = player2;
		
	BEGIN
		manager1 := 5;
		manager2 := 5;
		update player set managerid = manager2 where playerid = player1;
		update player set managerid = manager1 where playerid = player2;
	END;
	/
